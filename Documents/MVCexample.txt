class MonsterView {

    public void resize(Camera cam) {
        batch.setProjectionMatrix(camera.combined);
    }

    public void render(Vector2 position) {
        sprite.position = position;
        batch.begin();
        sprite.draw(batch);
        batch.end();
    }
}
